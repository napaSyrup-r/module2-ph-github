git add .

# 3. every time you commit 
git commit -m "first commit"

# 4. best practice for determining branch name 
git branch -M main

# 5. one time for each repo
git remote add origin https://github.com/napaSyrup-r/module2-ph-github.git

# 6.
git push -u origin main
